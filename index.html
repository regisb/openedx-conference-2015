def get_upload_url():
    conn = connect_to_s3()

Setting-dependent 
def get_upload_url():
    from importlib import import_module
    name = getattr(
        settings, "STORAGE_BACKEND",
        "storage.s3" # default storage
    )
    backend = import_module(name).Backend()
    return backend.generate_url()
